# Anita 项目依赖清单
# 本文件列出项目运行所需的所有依赖

# ==========================================
# 系统要求
# ==========================================
Node.js >= 20.0.0
npm >= 10.0.0
PostgreSQL >= 14.0

# ==========================================
# 后端核心依赖 (Backend)
# ==========================================
express@^4.18.2              # Web 框架
playwright@^1.40.1           # 浏览器自动化
pg@^8.11.3                   # PostgreSQL 客户端
@anthropic-ai/sdk@^0.71.0    # Claude AI SDK
typescript@^5.3.3            # TypeScript 编译器
dotenv@^16.3.1               # 环境变量管理
cors@^2.8.5                  # 跨域支持
axios@^1.6.2                 # HTTP 客户端
nodemailer@^7.0.11           # 邮件服务
node-cron@^4.2.1             # 定时任务
sharp@^0.33.1                # 图片处理
zod@^3.22.4                  # 数据验证

# ==========================================
# 前端核心依赖 (Frontend)
# ==========================================
react@^18.2.0                # React 框架
react-dom@^18.2.0            # React DOM
react-router-dom@^6.21.1     # React 路由
vite@^5.0.11                 # 构建工具
@tanstack/react-query@^5.17.0  # 数据获取
tailwindcss@^3.4.0           # CSS 框架
lucide-react@^0.555.0        # 图标库
recharts@^2.10.3             # 图表库

# ==========================================
# Docker 镜像 (可选，用于容器化部署)
# ==========================================
node:20-slim                 # 后端运行环境
nginx:alpine                 # 前端 Web 服务器
postgres:16-alpine           # 数据库

# ==========================================
# 外部服务 (可选)
# ==========================================
# SMTP 邮件服务 - 用于发送通知邮件
# Anthropic API - 用于 AI 性能分析

# ==========================================
# 安装命令
# ==========================================
# 完整安装: npm install
# 后端安装: cd backend && npm install
# 前端安装: cd frontend && npm install
# 浏览器安装: npx playwright install chromium
# Docker 部署: docker-compose up -d

# ==========================================
# 系统依赖 (Linux)
# ==========================================

# libcups2 libdrm2 libdbus-1-3 libxkbcommon0
# libatspi2.0-0 libxcomposite1 libxdamage1 libxfixes3
# libxrandr2 libgbm1 libpango-1.0-0 libcairo2
# libasound2 libxshmfence1 fonts-noto-color-emoji

# 硬件推荐配置
# ==========================================

# 内存: 8GB 及以上
# 磁盘: 20GB 可用空间
